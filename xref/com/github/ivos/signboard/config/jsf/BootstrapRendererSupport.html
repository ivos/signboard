18  					null);

22  		}

24  			writer.writeAttribute("class", "icon-ok-sign icon-large", null);
25  		}

36  		if (FacesMessage.SEVERITY_ERROR == message.getSeverity()
37  				|| FacesMessage.SEVERITY_FATAL == message.getSeverity()) {

39  		}
40  		if (FacesMessage.SEVERITY_INFO == message.getSeverity()) {
41  			styleClass.append(" alert-success");
