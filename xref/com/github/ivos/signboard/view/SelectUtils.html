28  	private <E extends Enum<E>> List<SelectItem> convertToSelectItems(
29  			Class<E> enumClass, boolean addEmptyItem) {
30  		List<SelectItem> list = new ArrayList<SelectItem>();
31  		if (addEmptyItem) {
32  			list.add(new SelectItem(null, ""));
33  		}
34  		Enum<?>[] enums = enumClass.getEnumConstants();
35  		for (Enum<?> e : enums) {
36  			String code = "enum_" + enumClass.getSimpleName() + "_"
37  					+ e.toString();
38  			list.add(new SelectItem(e, viewContext.getLabel(code)));
39  		}
40  		return list;
41  	}
